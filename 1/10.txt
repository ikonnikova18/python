print("Добро пожаловать! Вы заблудились, перед вами 3 двери, в какую направитесь: красная, желтая или зеленая.")
while True:
    a = input()
    if a == "красная":
        print("Вы упали в бездну. Игра окончена. Попробуйте еще раз.")
        break
    elif a == "желтая":
        print("Вас съел дракон! Игра окончена. Попробуйте еще раз.")
        break
    elif a == "зеленая":
        print("Поздравляю, вы прошли дальше и попали в пещеру. Есть два тунеля, ваш выбор: справа или слева?")
        b = input()
        if b == "справа":
            print("Вы встретились с ведьмой, которая превратила вас в камень. Попробуйте еще раз.")
            break
        elif b == "слева":
            print("Вы вышли в цивилизацию и прошли игру!")
            break
