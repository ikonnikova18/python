spisok = list(input("Введите любой список букв/чисел без пробелов: "))
print()

print("Будет выведен отсортированный по алфавиту/возрастанию список ")
print("Если хотите заменить данный список нажмите 1, если хотите создать новый нажмите 0")
a = int(input())    # переменная для определения какой список вывести
b = list()  # второй список для записи новых данных


if a == 1:
    spisok.sort()   # Метод sort() сортирует список по алфавиту/возрастанию, при этом перезаписывая данные в списке
    print("Исходный список с новыми данными: ", spisok)
    
else:
    b = sorted(spisok)  # Метод sorted() сортирует список по алфавиту/возрастанию, при этом записывая данные в новый список и не трогает данные в исходном
    print("Исходный список: ", spisok)
    print("Новый список: ", b)
